🤖 AI 챗봇 아키텍처

고객이 자연어로 메뉴를 질문하면, 백엔드가 실시간으로 DB에서 전체 메뉴를 조회해 AI 시스템 프롬프트에 주입합니다. 멀티턴 대화 히스토리를 유지하며 문맥에 맞는 추천을 제공합니다.

# routers/chat.py — 핵심 로직 요약
@router.post("/chat")
def chat(body, db):
  store_name, menu_text = _build_menu_context(db) # DB 메뉴 조회
  system = f"당신은 '{store_name}' 레스토랑 AI 도우미... {menu_text}"
  messages = [system] + history + [user_message]
  client = OpenAI(api_key=GROQ_API_KEY, base_url="https://api.groq.com/openai/v1")
  response = client.chat.completions.create(model="llama-3.3-70b-versatile", ...)
  return {"reply": response.choices[0].message.content}
